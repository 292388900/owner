11  import java.io.IOException;
12  import java.net.URL;
13  import java.net.URLConnection;
14  import java.net.URLStreamHandler;
15  
16  /**
17   * Handler for URLs used to indicate {@link Config.Sources Sources} from which to load the properties.
18   * It implements also system variables expander, using values from {@link System#getProperties()} and
19   * {@link System#getenv()}.
20   * The <tt>~</tt> (tilde) character is expanded as <tt>${user.home}</tt>.
21   *
22   * Some sample of valid URL handled by this class are:
23   * <ul>
24   *     <li>classpath:com.foo.bar.FooBar.properties</li>
25   *     <li>file:${user.dir}/src/test/resources/test.properties</li>
26   * </ul>
27   *
28   * This class implements by its own the classpath: protocol, other protocols are handled by delegating to default by
29   * default {@link java.net.URL#URL(String, String, int, String) URL} class.
30   * So you can also use http or ftp - for instance - to indicate the {@link Config.Sources Sources} for a property to
31   * load, as well as any other supported by {@link java.net.URL} class.
32   *
33   * @author Luigi R. Viggiano
34   */
35  class ConfigURLStreamHandler extends URLStreamHandler {
36      private final ClassLoader classLoader;
37      private final SystemVariablesExpander expander;
38  
39      public ConfigURLStreamHandler(ClassLoader classLoader, SystemVariablesExpander expander) {
40          this.classLoader = classLoader;
41          this.expander = expander;
42      }
43  
44      @Override
45      protected URLConnection openConnection(URL url) throws IOException {
46          String protocol = url.getProtocol();
47          String path = url.getPath();
48          if ("classpath".equals(protocol)) {
49              URL resourceUrl = classLoader.getResource(expand(path));
50              if (resourceUrl == null) return null;
51              return resourceUrl.openConnection();
52          } else {
53              return new URL(url.getProtocol(), url.getHost(), url.getPort(), expand(path)).openConnection();
54          }
55      }
56  
57      private String expand(String path) {
58          return expander.expand(path);
59      }
60  }
