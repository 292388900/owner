12  import java.util.LinkedHashMap;
13  import java.util.Map;
14  import java.util.Map.Entry;
15  import java.util.Properties;
16  import java.util.Set;

18  /**
19   * This class is used to expand variables in the format <tt>${variable}</tt>$, using values from {@link System#getenv()}
20   * and {@link System#getProperties()} (in this order; first match is accepted).
21   *
22   * @author Luigi R. Viggiano
23   */
24  class SystemVariablesExpander {
25  
26      private final StrSubstitutor substitutor;
27  
28      SystemVariablesExpander() {
29          Map<String, String> variables = new LinkedHashMap<String, String>(System.getenv());
30          addAll(variables, System.getProperties());
31          substitutor = new StrSubstitutor(variables);
32      }
33  
34      private void addAll(Map<String, String> variables, Properties properties) {
35          Set<Entry<Object, Object>> entries = properties.entrySet();
36          for (Entry<Object, Object> entry : entries)
37              variables.put((String) entry.getKey(), (String) entry.getValue());
38      }
39  
40      String expand(String path) {
41          if (path.indexOf('~') != -1)
42              path = path.replace("~", "${user.home}");
43          return substitutor.replace(path);
44      }
45  }
