12  import org.apache.commons.lang.text.StrSubstitutor;
13  
14  import java.util.LinkedHashMap;
15  import java.util.Map;
16  
17  /**
18   * This class is used to expand variables in the format <tt>${variable}</tt>$, using values from {@link System#getenv()}
19   * and {@link System#getProperties()} (in this order; first match is accepted).
20   *
21   * @author Luigi R. Viggiano
22   */
23  class SystemVariablesExpander {
24  
25      private final StrSubstitutor substitutor;
26  
27      SystemVariablesExpander() {
28          Map<Object, Object> properties = new LinkedHashMap<Object, Object>(System.getenv());
29          properties.putAll(System.getProperties());
30          substitutor = new StrSubstitutor(properties);
31      }
32  
33      String expand(String path) {
34          if (path.indexOf('~') != -1)
35              path = path.replace("~", "${user.home}");
36          return substitutor.replace(path);
37      }
38  }
