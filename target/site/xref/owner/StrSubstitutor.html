10  import java.util.Map;
11  import java.util.regex.Matcher;
12  import java.util.regex.Pattern;
13  
14  import static java.util.regex.Pattern.compile;
15  
16  /**
17   * Substitutes variables within a string by values.
18   * <p>
19   * This class takes a piece of text and substitutes all the variables within it.
20   * The definition of a variable is <code>${variableName}</code>.
21   * <p>
22   * Typical usage of this class follows the following pattern: First an instance is created
23   * and initialized with the valueMap that contains the values for the available variables.
24   * If a prefix and/or suffix for variables should be used other than the default ones,
25   * the appropriate settings can be performed. After that the <code>replace()</code>
26   * method can be called passing in the source text for interpolation. In the returned
27   * text all variable references (as long as their values are known) will be resolved.
28   * The following example demonstrates this:
29   * <pre>
30   * Map valuesMap = new HashMap();
31   * valuesMap.put(&quot;animal&quot;, &quot;quick brown fox&quot;);
32   * valuesMap.put(&quot;target&quot;, &quot;lazy dog&quot;);
33   * String templateString = &quot;The ${animal} jumped over the ${target}.&quot;;
34   * StrSubstitutor sub = new StrSubstitutor(valuesMap);
35   * String resolvedString = sub.replace(templateString);
36   * </pre>
37   * yielding:
38   * <pre>
39   *      The quick brown fox jumped over the lazy dog.
40   * </pre>
41   *
42   * @author Luigi R. Viggiano
43   */
44  class StrSubstitutor {
45      private Map<String, String> valueMap;
46      private static final Pattern PATTERN = compile("\\$\\{(.+?)\\}");
47  
48      /**
49       * Creates a new instance and initializes it. Uses defaults for variable
50       * prefix and suffix and the escaping character.
51       *
52       * @param valueMap  the valueMap with the variables' values, may be null
53       */
54      public StrSubstitutor(Map<String, String> valueMap) {
55          this.valueMap = valueMap;
56      }
57  
58      /**
59       * Replaces all the occurrences of variables with their matching values
60       * from the resolver using the given source string as a template.
61       *
62       * @param source  the string to replace in, null returns null
63       * @return the result of the replace operation
64       */
65      public String replace(String source) {
66          if (source == null) return null;
67          Matcher m = PATTERN.matcher(source);
68          StringBuffer sb = new StringBuffer();
69          while (m.find()) {
70              String var = m.group(1);
71              String replacement = valueMap.get(var);
72              m.appendReplacement(sb, replacement);
73          }
74          m.appendTail(sb);
75          return sb.toString();
76      }
77  }
